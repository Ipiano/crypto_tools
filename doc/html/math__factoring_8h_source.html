461 
462  //Guaranteed to be at least one factor
463  //Check if n might be of the form 2p^k
464  if(factors[0] == 2)
465  {
466  DBGOUT("Maybe 2p^k");
467  pk2 = true;
468  //If 2 is a factor, there's guaranteed to be at least some other factor
469  //because n is not 2
470  factors.erase(factors.begin());
471  }
472

479  for(const Integral& i : factors)
480  {
481  if(i != p) return false;
482  }
483 
484  //0 is never a primitive root
485  if(mod(a, p) == 0) return false;
486 
487  //Check if a is a primitive root mod the prime factor p of n
488 
489  //Factor p-1
490  Integral p1 = p-1;

494  {
